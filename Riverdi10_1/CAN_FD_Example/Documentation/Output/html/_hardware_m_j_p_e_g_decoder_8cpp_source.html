174bool HardwareMJPEGDecoder::decodeNextFrame(uint8_t* buffer, uint16_t buffer_width, uint16_t buffer_height, uint32_t buffer_stride)
175{
176 assert((frameNumber > 0) && "HardwareMJPEGDecoder decoding without frame data!");
177
178 /* find next frame and decode it */
179 readData(currentMovieOffset, 8);
180 uint32_t streamNo = getU16(currentMovieOffset);
181 uint32_t chunkType = getU16(currentMovieOffset + 2);
182 uint32_t chunkSize = getU32(currentMovieOffset + 4);
183
184 const uint16_t STREAM0 = 0x3030;
185 const uint16_t TYPEDC = 0x6364;
186
187 bool isCurrentFrameLast;
188 /* play frame if we have it all */
189 if (currentMovieOffset + 8 + chunkSize < movieLength)
190 {
191 if (streamNo == STREAM0 && chunkType == TYPEDC && chunkSize > 0)
192 {
193 currentMovieOffset += 8;
194 /* decode frame */
195 const uint8_t* chunk = readData(currentMovieOffset, chunkSize);
196 decodeMJPEGFrame(chunk, chunkSize, buffer, buffer_width, buffer_height, buffer_stride);
197 frameNumber++;
198 }
199
200 isCurrentFrameLast = false;
201
202 /* Advance to next frame */
203 currentMovieOffset += chunkSize;
204 if (chunkSize == 0) /* Empty frame - Skip */
205 {
206 currentMovieOffset += 8;
207 }
208 currentMovieOffset = (currentMovieOffset + 1) & 0xFFFFFFFE; /* pad to next word */
209
210 if (currentMovieOffset == lastFrameEnd)
211 {
212 frameNumber = 1;
213 currentMovieOffset = firstFrameOffset; /* start over */
214 isCurrentFrameLast = true;
215 }
216 }
217 else
218 {
219 frameNumber = 1;
220 currentMovieOffset = firstFrameOffset; /* start over */
221 isCurrentFrameLast = true;
222 }
223 return !isCurrentFrameLast;
224}

226bool HardwareMJPEGDecoder::gotoNextFrame()
227{
228 assert((frameNumber > 0) && "HardwareMJPEGDecoder decoding without frame data!");
229
230 readData(currentMovieOffset, 8);
231 uint32_t chunkSize = getU32(currentMovieOffset + 4);
232
233 /* increment until next video frame */
234 while (currentMovieOffset + 8 + chunkSize < movieLength)
235 {
236 /* increment one frame */
237 currentMovieOffset += chunkSize + 8;
238 currentMovieOffset = (currentMovieOffset + 1) & 0xFFFFFFFE; /* pad to next word */
239 frameNumber++;
240
241 /* next chunk */
242 readData(currentMovieOffset, 8);
243 /* check it is a video frame */
244 uint32_t streamNo = getU16(currentMovieOffset);
245 uint32_t chunkType = getU16(currentMovieOffset + 2);
246 chunkSize = getU32(currentMovieOffset + 4);
247 const uint16_t STREAM0 = 0x3030;
248 const uint16_t TYPEDC = 0x6364;
249
250 if (streamNo == STREAM0 && chunkType == TYPEDC)
251 {
252 /* Found next frame */
253 return true;
254 }
255 }
256
257 /* skip back to first frame */
258 frameNumber = 1;
259 currentMovieOffset = firstFrameOffset; /* start over */
260 return false;
261}

286void HardwareMJPEGDecoder::readVideoHeader()
287{
288 /* Start from the start */
289 currentMovieOffset = 0;
290 lastError = AVI_NO_ERROR;
291
292 /* Make header available in buffer */
293 readData(0, 72);
294
295 /* Decode the movie header to find first frame */
296 /* Must be RIFF file */
297 if (compare(currentMovieOffset, "RIFF", 4))
298 {
299 lastError = AVI_ERROR_NOT_RIFF;
300 assert(!"RIFF header not found");
301 }
302
303 /* skip fourcc and length */
304 currentMovieOffset += 8;
305 if (compare(currentMovieOffset, "AVI ", 4))
306 {
307 lastError = AVI_ERROR_AVI_HEADER_NOT_FOUND;
308 assert(!"AVI header not found");
309 }
310
311 currentMovieOffset += 4;
312 if (compare(currentMovieOffset, "LIST", 4))
313 {
314 lastError = AVI_ERROR_AVI_LIST_NOT_FOUND;
315 assert(!"AVI LIST not found");
316 }
317
318 /* save AVI List info */
319 const uint32_t aviListSize = getU32(currentMovieOffset + 4);
320 const uint32_t aviListOffset = currentMovieOffset;
321 assert(aviListSize);
322
323 /* look into header to find frame rate */
324 bool foundFrame = true;
325 uint32_t offset = currentMovieOffset + 8;
326 if (compare(offset, "hdrl", 4))
327 {
328 lastError = AVI_ERROR_AVI_HDRL_NOT_FOUND;
329 foundFrame = false;
330 }
331
332 offset += 4;
333 if (compare(offset, "avih", 4))
334 {
335 lastError = AVI_ERROR_AVI_AVIH_NOT_FOUND;
336 foundFrame = false;
337 }
338
339 if (foundFrame)
340 {
341 offset += 8; /* skip fourcc and cb in AVIMAINHEADER */
342 videoInfo.ms_between_frames = getU32(offset) / 1000;
343 videoInfo.number_of_frames = getU32(offset + 16);
344 videoInfo.frame_width = getU32(offset + 32);
345 videoInfo.frame_height = getU32(offset + 36);
346 }
347 /* skip rest of AVI header, start from end of AVI List */
348
349 /* look for list with 'movi' header */
350 uint32_t listOffset = aviListOffset + aviListSize + 8;
351 readData(listOffset, 12);
352 while (compare(listOffset + 8, "movi", 4) && (lastError == AVI_NO_ERROR) && listOffset < movieLength)
353 {
354 const uint32_t listSize = getU32(listOffset + 4) + 8;
355 listOffset += listSize;
356 readData(listOffset, 12);
357 }
358
359 if (lastError != AVI_NO_ERROR)
360 {
361 lastError = AVI_ERROR_MOVI_NOT_FOUND;
362 return;
363 }
364
365 /* save first frame and end of last frame */
366 currentMovieOffset = listOffset + 8 + 4; /* skip LIST and 'movi' */
367 lastFrameEnd = listOffset + 8 + getU32(listOffset + 4);
368
369 /* find idx */
370 const uint32_t listSize = getU32(listOffset + 4) + 8;
371 listOffset += listSize;
372 readData(listOffset, 4);
373 if (!compare(listOffset, "idx1", 4))
374 {
375 indexOffset = listOffset;
376 }
377 else
378 {
379 lastError = AVI_ERROR_IDX1_NOT_FOUND;
380 return;
381 }
382
383 /* start on first frame */
384 frameNumber = 1; /* next frame number is 1 */
385 firstFrameOffset = currentMovieOffset;
386}

388void HardwareMJPEGDecoder::decodeMJPEGFrame(const uint8_t* const mjpgdata, const uint32_t length, uint8_t* outputBuffer, uint16_t bufferWidth, uint16_t bufferHeight, uint32_t bufferStride)
389{
390 if (length == 0)
391 {
392 return;
393 }
394
395 if (outputBuffer) /* only decode if buffers are assigned. */
396 {
397 /* Update JPEG conversion parameters */
398 JPEG_ConvertorParams.bytes_pr_pixel = 2;
399 JPEG_ConvertorParams.WidthExtend = videoInfo.frame_width;
400 if ((JPEG_ConvertorParams.WidthExtend % 16) != 0)
401 {
402 JPEG_ConvertorParams.WidthExtend += 16 - (JPEG_ConvertorParams.WidthExtend % 16);
403 }
404 JPEG_ConvertorParams.ScaledWidth = 640 * JPEG_ConvertorParams.bytes_pr_pixel;
405 JPEG_ConvertorParams.MCU_pr_line = JPEG_ConvertorParams.WidthExtend / MCU_WIDTH_PIXELS;
406 JPEG_ConvertorParams.LastLineHeight = (videoInfo.frame_height % MCU_HEIGHT_PIXELS) == 0 ? 0 : MCU_HEIGHT_PIXELS - (videoInfo.frame_height % MCU_HEIGHT_PIXELS);
407
408 JPEG_Decode_DMA(&hjpeg, const_cast<uint8_t*>(mjpgdata), length, outputBuffer);
409 DMA2D_reference = dma;
410 do
411 {
412 JpegProcessing_End = JPEG_OutputHandler(&hjpeg);
413
414 /* If nothing to do, allow other tasks */
415 if (JpegProcessing_End == 2)
416 {
417 SEM_WAIT(semDecodingDone);
418 }
419 } while (JpegProcessing_End != 1);
420
421 /* reset flag */
422 Jpeg_HWDecodingEnd = 0;
423 DMA2D_CopyBufferEnd = 0;
424 }

668uint32_t JPEG_OutputHandler(JPEG_HandleTypeDef* hjpeg)
669{
670 /* Decode frame complete */
671 if (Jpeg_HWDecodingEnd && DMA2D_CopyBufferEnd)
672 {
673 return 1;
674 }
675
676 /* Try to start DMA2D video transfer if next buffer if full */
677 if (!DMA2D_reference->isDMARunning() && (Jpeg_OUT_BufferTab[JPEG_OUT_Read_BufferIndex].State == JPEG_BUFFER_FULL) && (DMA2D_CopyBufferEnd == 0))
678 {
679 DMA2D_reference->start();
680 }
681
682 /* Start JPEG IP if paused and next buffer is empty */
683 if ((JPEG_output_is_paused == 1) && (Jpeg_OUT_BufferTab[JPEG_OUT_Write_BufferIndex].State == JPEG_BUFFER_EMPTY) && (Jpeg_HWDecodingEnd == 0))
684 {
685 JPEG_output_is_paused = 0;
686 HAL_JPEG_Resume(hjpeg, JPEG_PAUSE_RESUME_OUTPUT);
687 }
688
689 return 2;
690}

697void DMA2D_CopyBuffer(JPEG_Data_BufferTypeDef& job)
698{
699 uint32_t xRef, yRef, refline;
700
701 yRef = ((job.MCU_index * MCU_WIDTH_PIXELS) / JPEG_ConvertorParams.WidthExtend) * MCU_WIDTH_PIXELS;
702 xRef = ((job.MCU_index * MCU_WIDTH_PIXELS) % JPEG_ConvertorParams.WidthExtend);
703
704 refline = (JPEG_ConvertorParams.ScaledWidth) * yRef + (JPEG_ConvertorParams.bytes_pr_pixel * xRef);
705
706 /* DMA2D OPFCCR register configuration */
707 WRITE_REG(DMA2D->OPFCCR, DMA2D_OUTPUT_RGB565);
708
709 /* Configure DMA2D data size */
710 if (job.MCU_index >= (MCU_TotalNb - JPEG_ConvertorParams.MCU_pr_line)) /* Last line of frame */
711 {
712 WRITE_REG(DMA2D->NLR, (MCU_HEIGHT_PIXELS - JPEG_ConvertorParams.LastLineHeight | ((MCU_WIDTH_PIXELS * JPEG_ConvertorParams.MCU_pr_line) << DMA2D_NLR_PL_Pos)));
713 }
714 else
715 {
716 WRITE_REG(DMA2D->NLR, (MCU_HEIGHT_PIXELS | ((MCU_WIDTH_PIXELS * JPEG_ConvertorParams.MCU_pr_line) << DMA2D_NLR_PL_Pos)));
717 }
718
719 /* Configure DMA2D destination address */
720 WRITE_REG(DMA2D->OMAR, (reinterpret_cast<uint32_t>(FrameBufferAddress) + refline));
721
722 /* DMA2D OOR register configuration */
723 WRITE_REG(DMA2D->OOR, 640 - (MCU_WIDTH_PIXELS * JPEG_ConvertorParams.MCU_pr_line));
724
725 /* DMA2D FGOR register configuration */
726 WRITE_REG(DMA2D->FGOR, 0);
727
728 /* DMA2D FGPFCCR register configuration */
729 WRITE_REG(DMA2D->FGPFCCR, DMA2D_INPUT_YCBCR | (DMA2D_CSS_420 << DMA2D_FGPFCCR_CSS_Pos) | (DMA2D_REPLACE_ALPHA << DMA2D_BGPFCCR_AM_Pos) | (0xFFU << DMA2D_FGPFCCR_ALPHA_Pos));
730
731 /* Configure DMA2D source address */
732 WRITE_REG(DMA2D->FGMAR, reinterpret_cast<uint32_t>(job.DataBuffer));
733
734 /* Configure DMA2D contol register */
735 WRITE_REG(DMA2D->CR, DMA2D_M2M_PFC | DMA2D_IT_TC | DMA2D_CR_START);
736}

743void DMA2D_ExternalJobCompleted(JPEG_Data_BufferTypeDef& job)
744{
