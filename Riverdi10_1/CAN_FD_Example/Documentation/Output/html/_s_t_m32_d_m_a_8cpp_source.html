38extern "C" void DMA2D_IRQHandler()
39{
40 /* Transfer Complete Interrupt management ************************************/
41 if ((READ_REG(DMA2D->ISR) & DMA2D_FLAG_TC) != RESET)
42 {
43 /* Verify Transfer Complete Interrupt */
44 if ((READ_REG(DMA2D->CR) & DMA2D_IT_TC) != RESET)
45 {
46 /* Disable the transfer complete interrupt */
47 DMA2D->CR &= ~(DMA2D_IT_TC);
48
49 /* Clear the transfer complete flag */
50 DMA2D->IFCR = (DMA2D_FLAG_TC);
51
52 /* Signal DMA queue of execution complete */
53 touchgfx::HAL::getInstance()->signalDMAInterrupt();
54 }
55 }
56}

82inline uint32_t STM32DMA::getChromARTInputFormat(Bitmap::BitmapFormat format)
83{
84 // Default color mode set to ARGB8888
85 uint32_t dma2dColorMode = DMA2D_INPUT_ARGB8888;
86
87 switch (format)
88 {
89 case Bitmap::ARGB8888: /* DMA2D input mode set to 32bit ARGB */
90 dma2dColorMode = DMA2D_INPUT_ARGB8888;
91 break;
92 case Bitmap::RGB888: /* DMA2D input mode set to 24bit RGB */
93 dma2dColorMode = DMA2D_INPUT_RGB888;
94 break;
95 case Bitmap::RGB565: /* DMA2D input mode set to 16bit RGB */
96 dma2dColorMode = DMA2D_INPUT_RGB565;
97 break;
98 case Bitmap::ARGB2222: /* Fall through */
99 case Bitmap::ABGR2222: /* Fall through */
100 case Bitmap::RGBA2222: /* Fall through */
101 case Bitmap::BGRA2222: /* Fall through */
102 case Bitmap::L8: /* DMA2D input mode set to 8bit Color Look up table*/
103 dma2dColorMode = DMA2D_INPUT_L8;
104 break;
105 case Bitmap::BW: /* Fall through */
106 case Bitmap::BW_RLE: /* Fall through */
107 case Bitmap::GRAY4: /* Fall through */
108 case Bitmap::GRAY2: /* Fall through */
109 default: /* Unsupported input format for DMA2D */
110 assert(0 && "Unsupported Format!");
111 break;
112 }
113
114 return dma2dColorMode;
115}

163/*
164 * void STM32DMA::setupDataCopy(const BlitOp& blitOp) handles blit operation of
165 * BLIT_OP_COPY
166 * BLIT_OP_COPY_L8
167 * BLIT_OP_COPY_WITH_ALPHA
168 * BLIT_OP_COPY_ARGB8888
169 * BLIT_OP_COPY_ARGB8888_WITH_ALPHA
170 * BLIT_OP_COPY_A4
171 * BLIT_OP_COPY_A8
172 */
173void STM32DMA::setupDataCopy(const BlitOp& blitOp)
174{
175 uint32_t dma2dForegroundColorMode = getChromARTInputFormat(static_cast<Bitmap::BitmapFormat>(blitOp.srcFormat));
176 uint32_t dma2dBackgroundColorMode = getChromARTInputFormat(static_cast<Bitmap::BitmapFormat>(blitOp.dstFormat));
177 uint32_t dma2dOutputColorMode = getChromARTOutputFormat(static_cast<Bitmap::BitmapFormat>(blitOp.dstFormat));
178
179 /* DMA2D OOR register configuration */
180 WRITE_REG(DMA2D->OOR, blitOp.dstLoopStride - blitOp.nSteps);
181
182 /* DMA2D BGOR register configuration */
183 WRITE_REG(DMA2D->BGOR, blitOp.dstLoopStride - blitOp.nSteps);
184
185 /* DMA2D FGOR register configuration */
186 WRITE_REG(DMA2D->FGOR, blitOp.srcLoopStride - blitOp.nSteps);
187
188 /* DMA2D OPFCCR register configuration */
189 WRITE_REG(DMA2D->OPFCCR, dma2dOutputColorMode);
190
191 /* Configure DMA2D data size */
192 WRITE_REG(DMA2D->NLR, (blitOp.nLoops | (blitOp.nSteps << DMA2D_NLR_PL_Pos)));
193
194 /* Configure DMA2D destination address */
195 WRITE_REG(DMA2D->OMAR, reinterpret_cast<uint32_t>(blitOp.pDst));
196
197 /* Configure DMA2D source address */
198 WRITE_REG(DMA2D->FGMAR, reinterpret_cast<uint32_t>(blitOp.pSrc));
199
200 switch (blitOp.operation)
201 {
202 case BLIT_OP_COPY_A4:
203 /* Set DMA2D color mode and alpha mode */
204 WRITE_REG(DMA2D->FGPFCCR, DMA2D_INPUT_A4 | (DMA2D_COMBINE_ALPHA << DMA2D_FGPFCCR_AM_Pos) | (blitOp.alpha << 24));
205
206 /* set DMA2D foreground color */
207 WRITE_REG(DMA2D->FGCOLR, blitOp.color);
208
209 /* Write DMA2D BGPFCCR register */
210 WRITE_REG(DMA2D->BGPFCCR, dma2dBackgroundColorMode | (DMA2D_NO_MODIF_ALPHA << DMA2D_BGPFCCR_AM_Pos));
211
212 /* Configure DMA2D Stream source2 address */
213 WRITE_REG(DMA2D->BGMAR, reinterpret_cast<uint32_t>(blitOp.pDst));
214
215 /* Set DMA2D mode */
216 WRITE_REG(DMA2D->CR, DMA2D_M2M_BLEND | DMA2D_IT_TC | DMA2D_CR_START);
217 break;
218 case BLIT_OP_COPY_A8:
219 /* Set DMA2D color mode and alpha mode */
220 WRITE_REG(DMA2D->FGPFCCR, DMA2D_INPUT_A8 | (DMA2D_COMBINE_ALPHA << DMA2D_FGPFCCR_AM_Pos) | (blitOp.alpha << 24));
221
222 /* set DMA2D foreground color */
223 WRITE_REG(DMA2D->FGCOLR, blitOp.color);
224
225 /* Write DMA2D BGPFCCR register */
226 WRITE_REG(DMA2D->BGPFCCR, dma2dBackgroundColorMode | (DMA2D_NO_MODIF_ALPHA << DMA2D_BGPFCCR_AM_Pos));
227
228 /* Configure DMA2D Stream source2 address */
229 WRITE_REG(DMA2D->BGMAR, reinterpret_cast<uint32_t>(blitOp.pDst));
230
231 /* Set DMA2D mode */
232 WRITE_REG(DMA2D->CR, DMA2D_M2M_BLEND | DMA2D_IT_TC | DMA2D_CR_START);
233 break;
234 case BLIT_OP_COPY_WITH_ALPHA:
235 /* Set DMA2D color mode and alpha mode */
236 WRITE_REG(DMA2D->FGPFCCR, dma2dForegroundColorMode | (DMA2D_COMBINE_ALPHA << DMA2D_FGPFCCR_AM_Pos) | (blitOp.alpha << 24));
237
238 /* Write DMA2D BGPFCCR register */
239 WRITE_REG(DMA2D->BGPFCCR, dma2dBackgroundColorMode | (DMA2D_NO_MODIF_ALPHA << DMA2D_BGPFCCR_AM_Pos));
240
241 /* Configure DMA2D Stream source2 address */
242 WRITE_REG(DMA2D->BGMAR, reinterpret_cast<uint32_t>(blitOp.pDst));
243
244 /* Set DMA2D mode */
245 WRITE_REG(DMA2D->CR, DMA2D_M2M_BLEND | DMA2D_IT_TC | DMA2D_CR_START);
246 break;
247 case BLIT_OP_COPY_L8:
248 {
249 bool blend = true;
250 const clutData_t* const palette = reinterpret_cast<const clutData_t*>(blitOp.pClut);
251
252 /* Write foreground CLUT memory address */
253 WRITE_REG(DMA2D->FGCMAR, reinterpret_cast<uint32_t>(&palette->data));
254
255 /* Set DMA2D color mode and alpha mode */
256 WRITE_REG(DMA2D->FGPFCCR, dma2dForegroundColorMode | (DMA2D_COMBINE_ALPHA << DMA2D_FGPFCCR_AM_Pos) | (blitOp.alpha << 24));
257
258 /* Write DMA2D BGPFCCR register */
259 WRITE_REG(DMA2D->BGPFCCR, dma2dBackgroundColorMode | (DMA2D_NO_MODIF_ALPHA << DMA2D_BGPFCCR_AM_Pos));
260
261 /* Configure DMA2D Stream source2 address */
262 WRITE_REG(DMA2D->BGMAR, reinterpret_cast<uint32_t>(blitOp.pDst));
263
264 /* Configure CLUT */
265 switch ((Bitmap::ClutFormat)palette->format)
266 {
267 case Bitmap::CLUT_FORMAT_L8_ARGB8888:
268 /* Write foreground CLUT size and CLUT color mode */
269 MODIFY_REG(DMA2D->FGPFCCR, (DMA2D_FGPFCCR_CS | DMA2D_FGPFCCR_CCM), (((palette->size - 1) << DMA2D_FGPFCCR_CS_Pos) | (DMA2D_CCM_ARGB8888 << DMA2D_FGPFCCR_CCM_Pos)));
270 break;
271 case Bitmap::CLUT_FORMAT_L8_RGB888:
272 if (blitOp.alpha == 255)
273 {
274 blend = false;
275 }
276 MODIFY_REG(DMA2D->FGPFCCR, (DMA2D_FGPFCCR_CS | DMA2D_FGPFCCR_CCM), (((palette->size - 1) << DMA2D_FGPFCCR_CS_Pos) | (DMA2D_CCM_RGB888 << DMA2D_FGPFCCR_CCM_Pos)));
277 break;
278
279 case Bitmap::CLUT_FORMAT_L8_RGB565:
280 default:
281 assert(0 && "Unsupported format");
282 break;
283 }
284
285 /* Enable the CLUT loading for the foreground */
286 SET_BIT(DMA2D->FGPFCCR, DMA2D_FGPFCCR_START);
287
288 while ((READ_REG(DMA2D->FGPFCCR) & DMA2D_FGPFCCR_START) != 0U)
289 {
290 }
291 DMA2D->IFCR = (DMA2D_FLAG_CTC);
292
293 /* Set DMA2D mode */
294 if (blend)
295 {
296 WRITE_REG(DMA2D->CR, DMA2D_M2M_BLEND | DMA2D_IT_TC | DMA2D_CR_START);
297 }
298 else
299 {
300 WRITE_REG(DMA2D->CR, DMA2D_M2M_PFC | DMA2D_IT_TC | DMA2D_CR_START);
301 }
302 }
303 break;
304 case BLIT_OP_COPY_ARGB8888:
305 case BLIT_OP_COPY_ARGB8888_WITH_ALPHA:
306 /* Set DMA2D color mode and alpha mode */
307 WRITE_REG(DMA2D->FGPFCCR, dma2dForegroundColorMode | (DMA2D_COMBINE_ALPHA << DMA2D_FGPFCCR_AM_Pos) | (blitOp.alpha << 24));
308
309 /* Write DMA2D BGPFCCR register */
310 WRITE_REG(DMA2D->BGPFCCR, dma2dBackgroundColorMode | (DMA2D_NO_MODIF_ALPHA << DMA2D_BGPFCCR_AM_Pos));
311
312 /* Configure DMA2D Stream source2 address */
313 WRITE_REG(DMA2D->BGMAR, reinterpret_cast<uint32_t>(blitOp.pDst));
314
315 /* Set DMA2D mode */
316 WRITE_REG(DMA2D->CR, DMA2D_M2M_BLEND | DMA2D_IT_TC | DMA2D_CR_START);
317 break;
318 default: /* BLIT_OP_COPY */
319 /* Set DMA2D color mode and alpha mode */
320 WRITE_REG(DMA2D->FGPFCCR, dma2dForegroundColorMode | (DMA2D_COMBINE_ALPHA << DMA2D_FGPFCCR_AM_Pos) | (blitOp.alpha << 24));
321
322 /* Perform pixel-format-conversion (PFC) If Bitmap format is not same format as framebuffer format */
323 if (blitOp.srcFormat != blitOp.dstFormat)
324 {
325 /* Start DMA2D : PFC Mode */
326 WRITE_REG(DMA2D->CR, DMA2D_M2M_PFC | DMA2D_IT_TC | DMA2D_CR_START);
327 }
328 else
329 {
330 /* Start DMA2D : M2M Mode */
331 WRITE_REG(DMA2D->CR, DMA2D_M2M | DMA2D_IT_TC | DMA2D_CR_START);
332 }
333 break;
334 }
335}

337/*
338 * void STM32DMA::setupDataFill(const BlitOp& blitOp) handles blit operation of
339 * BLIT_OP_FILL
340 * BLIT_OP_FILL_WITH_ALPHA
341 */
342void STM32DMA::setupDataFill(const BlitOp& blitOp)
343{
344 uint32_t dma2dOutputColorMode = getChromARTOutputFormat(static_cast<Bitmap::BitmapFormat>(blitOp.dstFormat));
345
346 /* DMA2D OPFCCR register configuration */
347 WRITE_REG(DMA2D->OPFCCR, dma2dOutputColorMode);
348
349 /* Configure DMA2D data size */
350 WRITE_REG(DMA2D->NLR, (blitOp.nLoops | (blitOp.nSteps << DMA2D_NLR_PL_Pos)));
351
352 /* Configure DMA2D destination address */
353 WRITE_REG(DMA2D->OMAR, reinterpret_cast<uint32_t>(blitOp.pDst));
354
355 /* DMA2D OOR register configuration */
356 WRITE_REG(DMA2D->OOR, blitOp.dstLoopStride - blitOp.nSteps);
357
358 if (blitOp.operation == BLIT_OP_FILL_WITH_ALPHA)
359 {
360 /* DMA2D BGOR register configuration */
361 WRITE_REG(DMA2D->BGOR, blitOp.dstLoopStride - blitOp.nSteps);
362
363 /* DMA2D FGOR register configuration */
364 WRITE_REG(DMA2D->FGOR, blitOp.dstLoopStride - blitOp.nSteps);
365
366 /* Write DMA2D BGPFCCR register */
367 WRITE_REG(DMA2D->BGPFCCR, dma2dOutputColorMode | (DMA2D_NO_MODIF_ALPHA << DMA2D_BGPFCCR_AM_Pos));
368
369 /* Write DMA2D FGPFCCR register */
370 WRITE_REG(DMA2D->FGPFCCR, DMA2D_INPUT_A8 | (DMA2D_REPLACE_ALPHA << DMA2D_FGPFCCR_AM_Pos) | ((blitOp.alpha << 24) & DMA2D_FGPFCCR_ALPHA));
371
372 /* DMA2D FGCOLR register configuration */
373 WRITE_REG(DMA2D->FGCOLR, blitOp.color);
374
375 /* Configure DMA2D Stream source2 address */
376 WRITE_REG(DMA2D->BGMAR, reinterpret_cast<uint32_t>(blitOp.pDst));
377
378 /* Configure DMA2D source address */
379 WRITE_REG(DMA2D->FGMAR, reinterpret_cast<uint32_t>(blitOp.pDst));
380
381 /* Enable the Peripheral and Enable the transfer complete interrupt */
382 WRITE_REG(DMA2D->CR, (DMA2D_IT_TC | DMA2D_CR_START | DMA2D_M2M_BLEND));
383 }
384 else
385 {
386 /* Write DMA2D FGPFCCR register */
387 WRITE_REG(DMA2D->FGPFCCR, dma2dOutputColorMode | (DMA2D_NO_MODIF_ALPHA << DMA2D_FGPFCCR_AM_Pos));
388
389 /* DMA2D FGOR register configuration */
390 WRITE_REG(DMA2D->FGOR, 0);
391
392 /* Set color */
393 WRITE_REG(DMA2D->OCOLR, ((blitOp.color >> 8) & 0xF800) | ((blitOp.color >> 5) & 0x07E0) | ((blitOp.color >> 3) & 0x001F));
394
395 /* Enable the Peripheral and Enable the transfer complete interrupt */
396 WRITE_REG(DMA2D->CR, (DMA2D_IT_TC | DMA2D_CR_START | DMA2D_R2M));
397 }

615 }
616

622 }
