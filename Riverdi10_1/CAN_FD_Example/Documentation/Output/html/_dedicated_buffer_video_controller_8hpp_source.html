158 virtual void setCommand(const Handle handle, Command cmd, uint32_t param)
159 {
160 // Running in UI thread
161
162 assert(handle < no_streams);
163 Stream& stream = streams[handle];
164
165 switch (cmd)
166 {
167 case PLAY:
168 // Cannot Play without movie
169 if (mjpegDecoders[handle]->hasVideo())
170 {
171 stream.isPlaying = true;
172 stream.hasMoreFrames = true; // Assume more frames, lowered on end
173 // Reset counters
174 stream.frameCount = 0;
175 stream.tickCount = 0;
176 }
177 break;
178 case PAUSE:
179 stream.isPlaying = false;
180 break;
181 case SEEK:
182 stream.seek_to_frame = param;
183 // Reset counters
184 stream.frameCount = 0;
185 stream.tickCount = 0;
186 break;
187 case SHOW:
188 stream.seek_to_frame = param;
189 stream.isShowingOneFrame = true;
190 // Reset counters
191 stream.frameCount = 0;
192 stream.tickCount = 0;
193 break;
194 case STOP:
195 stream.isPlaying = false;
196 stream.seek_to_frame = 1;
197 break;
198 case SET_REPEAT:
199 stream.repeat = (param > 0);
200 break;
201 }

204 virtual bool updateFrame(const Handle handle, touchgfx::VideoWidget& widget)
205 {
206 // Running in UI thread
207
208 assert(handle < no_streams);
209 Stream& stream = streams[handle];
210
211 // Increase tickCount if playing
212 if (stream.isPlaying)
213 {
214 stream.tickCount += HAL::getInstance()->getLCDRefreshCount();
215 }
216
217 // Invalidate Widget if new frame was decoded
218 if (stream.doInvalidateOnNextTick)
219 {
220 widget.setVideoBuffer((uint8_t*)decodeBuffers[handle]);
221 widget.invalidate();
222 stream.doInvalidateOnNextTick = false;
223 if (!stream.hasMoreFrames && !stream.repeat)
224 {
225 stream.isPlaying = false;
226 }
227 return stream.hasMoreFrames;
228 }
229 return true;
230 }

263 void endFrame()
264 {
265 // Running in UI thread
266
267 // Check if we should invalidate in next frame
268 for (uint32_t i = 0; i < no_streams; i++)
269 {
270 Stream& stream = streams[i];
271 if (stream.isPlaying || stream.isShowingOneFrame)
272 {
273 if (decodeForNextTick(stream))
274 {
275 stream.doDecode = true;
276 }
277 }
278 }

293 void decoderTaskEntry()
294 {
295 // Running in Decoder thread!!
296
297 while (1)
298 {
299 // Wait for synchronisation signal from UI thread
300 SEM_WAIT(semDecode);
301
302 // Lock out the UI by taking the mutex
303 MUTEX_LOCK(mutexBuffers);
304
305 // Update stream flags
306 for (uint32_t i = 0; i < no_streams; i++)
307 {
308 Stream& stream = streams[i];
309 if (stream.doDecode)
310 {
311 stream.doInvalidateOnNextTick = true;
312 }
313 }
314
315 // Now decode all streams marked for invalidation in next tick
316 for (uint32_t i = 0; i < no_streams; i++)
317 {
318 Stream& stream = streams[i];
319 if (stream.doDecode)
320 {
321 MJPEGDecoder* const decoder = mjpegDecoders[i];
322 // Seek or increment video frame
323 if (stream.seek_to_frame > 0)
324 {
325 decoder->gotoFrame(stream.seek_to_frame);
326 stream.seek_to_frame = 0;
327 }
328 else
329 {
330 if (stream.skip_frames > 0)
331 {
332 decoder->gotoFrame(decoder->getCurrentFrameNumber() + stream.skip_frames);
333 stream.frameCount += stream.skip_frames;
334 stream.skip_frames = 0;
335 }
336 }
337
338 // Decode frame
339 stream.hasMoreFrames = decoder->decodeNextFrame((uint8_t*)decodeBuffers[i], width, height, stride);
340 stream.frameCount++;
341
342 // Update frame number
343 if (stream.hasMoreFrames)
344 {
345 stream.frameNumber = decoder->getCurrentFrameNumber() - 1; // Points to frame after this
346 }
347 else
348 {
349 stream.frameNumber = 1;
350 }
351 stream.doDecode = false;
352 }
353 }
354
355 // Release the stream and decode buffer and let UI continue
356 MUTEX_UNLOCK(mutexBuffers);
357 }

418 bool decodeForNextTick(Stream& stream)
419 {
420 // Running in UI thread
421
422 // Compare tickCount/frameCount to frame_rate_ticks/frame_rate_video
423 if ((stream.tickCount * stream.frame_rate_video) >= (stream.frame_rate_ticks * stream.frameCount))
424 {
425 if (allowSkipFrames)
426 {
427 stream.skip_frames = (stream.tickCount * stream.frame_rate_video - stream.frame_rate_ticks * stream.frameCount) / stream.frame_rate_ticks;
428 if (stream.skip_frames > 0)
429 {
430 stream.skip_frames--;
431 }
432 }
433 return true;
434 }
435 return false;
436 }
