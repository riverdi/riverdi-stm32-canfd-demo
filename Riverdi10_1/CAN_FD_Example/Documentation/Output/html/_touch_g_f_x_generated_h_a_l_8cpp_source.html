179 void HAL_LTDC_LineEventCallback(LTDC_HandleTypeDef* hltdc)
180 {
181 if (!HAL::getInstance())
182 {
183 return;
184 }
185
186 if (LTDC->LIPCR == lcd_int_active_line)
187 {
188 //entering active area
189 HAL_LTDC_ProgramLineEvent(hltdc, lcd_int_porch_line);
190 HAL::getInstance()->vSync();
191 OSWrappers::signalVSync();
192
193 // Swap frame buffers immediately instead of waiting for the task to be scheduled in.
194 // Note: task will also swap when it wakes up, but that operation is guarded and will not have
195 // any effect if already swapped.
196 HAL::getInstance()->swapFrameBuffers();
197 GPIO::set(GPIO::VSYNC_FREQ);
198 }
199 else
200 {
201 //exiting active area
202 HAL_LTDC_ProgramLineEvent(hltdc, lcd_int_active_line);
203
204 // Signal to the framework that display update has finished.
205 HAL::getInstance()->frontPorchEntered();
206 GPIO::clear(GPIO::VSYNC_FREQ);
207 }
208 }
